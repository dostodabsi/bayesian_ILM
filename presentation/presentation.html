Markov Chains¶

Stochastic process whose present state $s_n$ does only depend on the preceeding $s_{n-1}$ but not on earlier states.

A state is expressed as a vector $S = (s_1, s_2)^T$. The transition from $S_n$ to $S_{n+1}$ is defined by the transition matrix $T$ which contains the transition probabilies for each of the elemts of $S_n$.

Example:

Consider two possible languages, $E$ and $A$. In each step the probability to change from $E$ to $A$ is 0.7 and to stay with $E$ is 0.3 and so on. This gives the transition matrix $T = \begin{pmatrix} 0.3 & 0.4 \\ 0.7 & 0.6 \end{pmatrix}$. For a given initial condition $S_0$ of the distribution of the languages $E$ and $A$ in a population the state $S_1$ is simply given by $T \cdot S_0$. The distribution in the $n$th generation equal to $T^n \cdot S_0$.

$S_0 = \begin{pmatrix} 0.9 \\ 0.1 \end{pmatrix}$ $\Rightarrow T \cdot S_0 = \begin{pmatrix} 0.31 \\ 0.69 \end{pmatrix} \Rightarrow T \cdot \begin{pmatrix} 0.31 \\ 0.69 \end{pmatrix} = \begin{pmatrix} 0.369 \\ 0.631 \end{pmatrix}$ ...

Simulation I: Early Work¶

Deriving the transition matrix for sampling over the posterior

$Q = \begin{pmatrix} q_{11} & q_{12} \\ q_{21} & q_{22} \end{pmatrix} = \begin{pmatrix} 1-q_{21} & q_{12} \\ q_{21} & 1-q_{12} \end{pmatrix}$

### The production algorithm PA

Production of an $(x,y)$ pair involves two stages (1) sampling an x according to $P(x)$ (2) producing an utterance $y$ appropriate to $x$ according to PA

$P_{PA}(d=(x,y)|h=i)\begin{cases}P(x)(1-\epsilon) & \text{if y maps to $x$ in $h_i$} \\ P(x)\epsilon & \mathrm{otherwise}\end{cases}$

### Using Bayes rule to derive the posterior

$P(h|d) = \frac{P_{PA}(d|h)P(h)}{\sum_{h\in H}P(d|h)P(h)}$

Let $\alpha = P(h_1)$, the prior.

Given a certain $(x,y)$ pair, according to PA, there are three cases $(x,y)$ can be the result of.

  1. $x$ belongs to the subset $S$ of utterances for which each $h$ produces the same $y$ $\Rightarrow \frac{(1-\epsilon)\alpha}{(1-\epsilon)\alpha + (1-\epsilon)(1-\alpha)} = \alpha$

  2. $x$ belongs to the subset $G_1$ for which only $h_1$ produces $y$ $\Rightarrow \frac{(1-\epsilon)\alpha}{(1-\epsilon)\alpha + \epsilon(1-\alpha)}$

  3. $x$ belongs to the subset $G_2$ for which only $h_2$ produces $y$ $\Rightarrow \frac{\epsilon\alpha}{\epsilon\alpha + (1-\epsilon)(1-\alpha)}$

### Determinig $q_{12}$ and $q_{21}$

$q_{12}$ detemines the portion of $h_2$ that switches to $h_1$ and $q_{21}$ the other way around. Thus , for the sampling algorithm, the values are the sums over the three cases, weighted with the respective probability that a observed $(x,y)$ pair belongs to one of the cases.

  1. $s = \sum_{x \in S}P(x)$
  2. $(1-s)(1-\epsilon)$ if $h_{n-1}$ was $h_1$; $(1-s)\epsilon$ if $h_{n-1}$ was $h_2$
  3. $(1-s)\epsilon$ if $h_{n-1}$ was $h_1$; $(1-s)(1-\epsilon)$ if $h_{n-1}$ was $h_2$

thus $q_{12} = s\alpha + \frac{(1-\epsilon)\alpha}{(1-\epsilon)\alpha + \epsilon(1-\alpha)}(1-s)\epsilon + \frac{\epsilon\alpha}{\epsilon\alpha + (1-\epsilon)(1-s)}(1-\epsilon)(1-\epsilon)$

since $q_{21}$ is the reverse process, it simply derived by interchaniging the $\alpha$ and $1-\alpha$.

function simulate(, ; n=10, s=.5, MAP=false)
    #  ~ prior;  ~ noise
    # s ~ probability for both languages producing the same y

    if MAP
        R(x) = x == .5 ? .5 : x > .5 ? 1 : 0
        q_12 = s + (1 - s)* + R(* / (* + (1 - )*(1 - ))) * (1 - s)*(1 - )
        q_21 = R(*(1 - ) / (*(1 - ) + (1 - )*)) * (1 - s)*(1 - )
    else
        q_12 = begin *s +
            ((1 - )* / ((1 - )* + *(1 - ))) * (1 - s)* +
            (* / (* + (1 - )*(1 - ))) * (1 - s)*(1 - )
        end

        q_21 = begin (1 - )*s +
            ((1 - )*(1 - ) / ((1 - )*(1 - ) + *)) * (1 - s)* +
            (*(1 - ) / (*(1 - ) + (1 - )*)) * (1 - s)*(1 - )
        end
    end

    Q = [1-q_21 q_12
         q_21 1-q_12]

    inits = MAP ? [.5 .5] : [1 0]
    S = [inits; zeros(n, 2)]

    for i in 2:n+1
        S[i, :] = Q * S[i-1, :]'
    end

    return S
end;

function simulate(;MAP=true, N_sam=1, N_gen=10000,
                   priors=[.8 .1 .1; .1 .1 .8], H=[.6 .2 .2; .2 .6 .2; .2 .2 .6])

    h_row, h_col = size(H)
    p_row, p_col = size(priors)
    @assert p_col == h_row

    N_pop = p_row # number of populations
    N_hyp = h_row # number of hypotheses
    N_dat = h_col # number of data points seen
    N_sampop = N_sam * N_pop # number of samples, agents per population

    # keep track of the history
    posterior_history = zeros(N_gen, N_pop, N_hyp)
    hypotheses_history = zeros(N_gen, N_pop)
    hyphist = zeros(N_pop, N_hyp)
    hyphistnorm = zeros(N_pop, N_hyp)

    posterior = zeros(N_pop, N_hyp)
    agents_posterior = zeros(N_pop, N_hyp)
    agents_hypothesis = zeros(N_pop, 1)
    posterior_mean = zeros(N_pop, N_hyp)

    # priors over hypotheses over populations
    priors = log(priors)

    # ~ likelihood of data points under each hypothesis
    hypotheses = log(H)

    # initial data matrix
    data = rand(1:N_dat, (1, N_sampop))

    # for each generation
    for generation in 1:N_gen

        # for each population
        for i in 1:N_pop
            likelihood = zeros(1, N_hyp) # reset the likelihood

            # for each sample per population per generation
            for j in 1:N_sampop
                likelihood += hypotheses[:, data[j]]' # hypotheses are the likelihoods
            end

            agents_posterior[i, :] = bayes_rule(priors[i, :], likelihood)
            posterior_history[generation, :, :] = agents_posterior
            posterior_mean[i, :] = sum(posterior_history[:, i, :]) ./ N_gen

            if MAP
                ############################
                post = agents_posterior[i, :]
                value = maximum(post)
                pos = findin(post, value)
                s = size(pos)[1]

                # if there is more than one maximum, choose randomly between them
                if s > 1
                    pos = pos[rand(1:s, 1)]
                end

                agents_hypothesis[i, :] = pos[1]
                hypotheses_history[generation, i] = pos[1]
                ####################################
            else # Sampling
                ############################
                weights = agents_posterior[i, :]'
                agents_hypothesis[i] = wsample(1:N_hyp, collect(weights), N_sam)[1]
                hypotheses_history[generation, i] = agents_hypothesis[i]
                ####################################
            end
        end

        # generate data
        data = Int64[]
        for i in 1:N_pop
            weights = exp(hypotheses[agents_hypothesis[i], :])
            agent_data = wsample(1:N_dat, collect(weights), N_sam)
            append!(data, agent_data)
        end
    end

    # create hypotheses history
    for pop in 1:N_pop
        for h in 1:N_hyp
            for g in 1:N_gen
                if hypotheses_history[g, pop] == h
                    hyphist[pop, h] += 1
                end
            end
        end
    end

    for i in 1:N_pop
        hyphistnorm[i, :] = hyphist[i, :] ./ sum(hyphist[i, :])
    end

    return hyphistnorm
end;

using DataFrames


function bayes(priors, hypotheses, data)
    llhs = mapslices(h -> *(h[d]...), hypotheses, 2) # P(d|h) for each h
    propto = llhs' .* priors # P(h) * P(d|h) for each h
    posterior = propto / sum(propto) # P(h|d) for each h
    DataFrame(data = rep(join(data), 2), prior = vec(prior),
         likelihood = vec(round(llhs, 4)), posterior = vec(round(posterior, 4)))
end;


function bayes_rule(prior, llh)
    posterior = exp(prior + llh)
    return posterior / sum(posterior)
end;

# simulate effect of bottleneck size
function simulate_bottlenecks(;MAP=true, canonical = true, bottlenecks=[1:10],
                               priors = [1/3 1/3 1/3])

    res = zeros(length(bottlenecks), 3)
    hypotheses = canonical ? [.6 .2 .2; .2 .6 .2; .2 .2 .6] :
                             [.6 .3 .1; .2 .6 .2; .1 .3 .6]

    for (i, b) in enumerate(bottlenecks)
        res[i, :] = simulate(priors = priors, H = hypotheses, N_sam = b)
    end
    
    return res
end

# compare effect of multiple {agents, samples} in a population
function simulate_population(;MAP = true, prior = [.7 .2 .1], agents = 4,
                              H = [.8 .1 .1; .1 .8 .1; .1 .1 .8])

    res = zeros(2, 3)
    bottleneck = agents
    multi_agent = vcat([prior for i in 1:agents]...)

    res[1, :] = simulate(priors = prior, H = H, N_sam = bottleneck, MAP = MAP)
    res[2, :] = mean(simulate(priors = multi_agent, H = H, MAP = MAP), 1)

    df = convert(DataFrame, round(res', 3))
    names!(df, [:population, :bottleneck])
    return df
end;
